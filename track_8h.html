uint8_t track_load_line ( uint8_t  _line) 
{
    uint8_t result = 0;

    if(_start_position + _line > TRACK_LENGTH)
    {
        result = LOAD(_start_position + _line - TRACK_LENGTH);
    }
    else
    {
        result = LOAD(_start_position + _line);
    }

    return result;
}
